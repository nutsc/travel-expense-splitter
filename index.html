üîí Security Setup Instructions:

  1. Go to Firebase Console ‚Üí Firestore Database ‚Üí Rules
  2. Replace with the rules shown below
  3. Change ACCESS_PASSWORD (for login)
  4. Change ADMIN_PASSWORD (for delete/clear)
  5. Click Publish
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Access control document
    match /travel/access {
      allow read: if true;
    }
    
    // Main data documents
    match /travel/{document} {
      // Everyone can read IF they know the access password
      allow read: if document != 'access';
      
      // Allow writes if:
      // 1. Password is empty (for normal adds), OR
      // 2. Password matches admin secret (for deletes/clears)
      allow write: if request.resource.data.get('_adminPassword', '') == '' 
                   || request.resource.data.get('_adminPassword', '') == 'ADMIN_PASSWORD';
    }
  }
}

// IMPORTANT: Create a document to store access password
// 1. Go to Firestore Database ‚Üí Data tab
// 2. Create collection: 'travel'
// 3. Create document with ID: 'access'
// 4. Add field: password (string) = 'ACCESS_PASSWORD'
